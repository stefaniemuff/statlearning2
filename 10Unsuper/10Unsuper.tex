\begin{frame}
\begin{block}{Proportion of varianced explained (PVE)}
\protect\hypertarget{proportion-of-varianced-explained-pve}{}
\(~\)

\textbf{Recap:} The PVE by PC \(m\) is given by

\[
\frac{\sum_{i=1}^m z_{im}^2} {\sum_{j=1}^p\sum_{i=1}^n x_{ij}^2}
\]
\end{block}
